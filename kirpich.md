```
# Задача про кирпич
# Кирпич вести 1 кг и еще пол кирпича. Сколько весит кирпич?
# Пусть x - вес кирпича
(0..100).each do |x| # Перебираем x от 0 до 100 кг
  if x==1+0.5*x # Если уравнение выполняется
    print "Кирпич весит ", x, "кг" # выводим результат
  end # конец if
end # конец цикла each
```
